    get_personalized_feed(self, user_id: Optional[int], limit: int = 50, cursor: Optional[str] = None) -> dict:
    - Check cache: Try GET feed:user:{user_id} from Redis (if user_id provided)
    - If cache hit and cursor is None: Parse cached list, return first {limit} items
    - Build feed:
      * If user_id is None (not authenticated): return get_trending_feed(limit, cursor)
      * Get followed users' IDs: FollowService.get_following(user_id, limit=100)
      * If no follows (count == 0): return get_trending_feed(limit, cursor) [cold start fallback]
      * Get followed notebooks: Query SELECT * FROM notebooks WHERE user_id IN ({followed_ids}) AND is_published=True AND is_archived=False ORDER BY created_at DESC LIMIT 50
      * Get trending notebooks: TrendingService.get_trending_notebooks(limit=50)
      * Merge and deduplicate: feed = followed_notebooks + [nb for nb in trending_notebooks if nb.id not in {fn.id for fn in followed_notebooks}]
      * Sort by: priority score (followed=2, trending=1) then created_at DESC
      * Apply cursor pagination if provided: filter notebooks where created_at < cursor
      * Limit to {limit} results
      * Determine next_cursor and has_more
    - Cache result: If user_id provided: LPUSH feed:user:{user_id} {notebook_ids}, EXPIRE feed:user:{user_id} 60 (1 min TTL)
    - For each notebook: record event if user_id provided (FeedEvent for future ML)
    - Return {"items": [NotebookResponse], "next_cursor": str, "has_more": bool}

    get_trending_feed(self, limit: int = 50, cursor: Optional[str] = None) -> dict:
    - Get trending notebook IDs: TrendingService.get_trending_notebooks(limit=limit + 1)
    - Fetch full notebooks: NotebookService.get_notebook_batch(ids)
    - Apply cursor pagination: filter where created_at < cursor
    - Limit to {limit} results
    - Determine next_cursor and has_more
    - Return {"items": [NotebookResponse], "next_cursor": str, "has_more": bool}

    invalidate_user_feed(self, user_id: int) -> None:
    - Delete feed cache for all followers of user_id
    - Query: SELECT follower_id FROM follows WHERE following_id = user_id
    - For each follower_id: DEL feed:user:{follower_id}
    - Per CONTEXT.md D-27: Cache invalidation on notebook publish/update

    get_engagement_metrics(self, notebook_ids: List[int]) -> dict:
    - Batch query engagement metrics from Redis hashes
    - For each notebook_id:
      * Get likes count: HGET notebook:{notebook_id}:score "likes" or COUNT from likes table
      * Get comments count: HGET notebook:{notebook_id}:score "comments" or COUNT from comments table
      * Get views count: HGET notebook:{notebook_id}:views "count" or 0
    - Return {notebook_id: {"likes": int, "comments": int, "views": int}}

    record_view(self, notebook_id: int, user_id: Optional[int] = None) -> None:
    - Increment view counter in Redis: HINCRBY notebook:{notebook_id}:views count 1
    - If user_id provided: Create FeedEvent(event_type="view")
    - Per CONTEXT.md D-31: View tracking via Redis, batch synced to DB

    Per CONTEXT.md D-12: Personalized feed = followed + trending, 0 follows = 100% trending
    Per CONTEXT.md D-23: Feed cache TTL 1 minute, lazy invalidation
    Per CONTEXT.md D-30: Engagement metrics displayed (likes, comments, views)
