    __init__(self, db: Session):
    - Initialize self.db = db
    - Initialize self.redis = get_redis_client()

    calculate_engagement_score(self, notebook_id: int) -> dict:
    - Query notebook with engagement: SELECT notebooks.id, COUNT(DISTINCT likes.id) as like_count, COUNT(DISTINCT comments.id) as comment_count FROM notebooks LEFT JOIN likes ON notebooks.id = likes.notebook_id LEFT JOIN comments ON notebooks.id = comments.notebook_id WHERE notebooks.id = notebook_id GROUP BY notebooks.id
    - Calculate engagement: engagement = (like_count * 2) + (comment_count * 3)
    - Calculate age_hours: (now() - notebooks.created_at).total_seconds() / 3600
    - Calculate decayed_score: engagement / pow((age_hours + 2), 1.5)
    - Return {"engagement": int, "decayed_score": float, "age_hours": float, "notebook_id": int}
    - Per CONTEXT.md D-11: engagement = (likes * 2) + (comments * 3) + (views * 0.05), decay exponent 1.5

    update_notebook_score(self, notebook_id: int) -> None:
    - Call calculate_engagement_score(notebook_id)
    - Store in Redis hash: HSET notebook:{notebook_id}:score {"engagement": engagement, "decayed_score": decayed_score, "updated_at": now().isoformat()}
    - Update trending ZSET: ZADD trending:all decayed_score "notebook:{notebook_id}"
    - Set expiration on hash: EXPIRE notebook:{notebook_id}:score 86400 (24 hours)

    increment_engagement(self, notebook_id: int, event_type: str) -> None:
    - Determine weight: like=2, comment=3, view=0.05
    - Increment engagement hash: HINCRBY notebook:{notebook_id}:score engagement {weight}
    - Get new engagement: HGET notebook:{notebook_id}:score engagement
    - Recalculate decayed score inline (fast operation)
    - Update ZSET: ZADD trending:all {decayed_score} "notebook:{notebook_id}"

    get_trending_notebooks(self, limit: int = 50) -> List[int]:
    - Query Redis ZSET in descending order: ZREVRANGE trending:all 0 {limit-1} WITHSCORES
    - Extract notebook IDs from returned list
    - Return list of notebook IDs (ints)

    recalculate_all_scores(self) -> None:
    - Query all published notebooks: SELECT id FROM notebooks WHERE is_published=True AND is_archived=False
    - For each notebook: call update_notebook_score(notebook_id)
    - Per CONTEXT.md D-25: Background recalculation every 2 minutes

    bootstrap_cache(self) -> None:
    - Check if trending:all ZSET exists: EXISTS trending:all
    - If exists, return (already bootstrapped)
    - Call recalculate_all_scores() to populate cache from database
    - Set flag: SET cache:bootstrapped "true" EX 86400
    - Per CONTEXT.md D-28: Bootstrap on deploy/run

    record_event(self, user_id: int, notebook_id: int, event_type: str, metadata: dict = None) -> None:
    - Create FeedEvent record in database (for future ML)
    - If event_type == "view": call increment_engagement(notebook_id, "view")
    - Per CONTEXT.md D-15: Event logging for ML foundation
  </action>
